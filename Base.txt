        function saveToPDF() {
            const { jsPDF } = window.jspdf;
            const pdf = new jsPDF();

            pdf.setFontSize(16);
            pdf.text("Safety Stop Card", 105, 10, { align: "center" });

            const date = document.getElementById("date").value;
            const time = document.getElementById("time").value;
            const location = document.getElementById("location").value;
            const observer = document.getElementById("observer").value;
            const position = document.getElementById("position").value;

            const description = document.getElementById("description").value;
            const immediateAction = document.getElementById("immediate-action").value;
            const recommendations = document.getElementById("recommendations").value;

            const unsafeActions = document.getElementById("unsafe-actions").checked;
            const unsafeConditions = document.getElementById("unsafe-conditions").checked;
            const safeActions = document.getElementById("safe-actions").checked;
            const safeConditions = document.getElementById("safe-conditions").checked;

            pdf.text(`Date: ${date}`, 10, 20);
            pdf.text(`Time: ${time}`, 10, 30);
            pdf.text(`Location: ${location}`, 10, 40);
            pdf.text(`Observer: ${observer}`, 10, 50);
            pdf.text(`Position: ${position}`, 10, 60);

            pdf.text("Observation:", 10, 70);
            if (unsafeActions) pdf.text("- Unsafe Actions", 20, 80);
            if (unsafeConditions) pdf.text("- Unsafe Conditions", 20, 90);
            if (safeActions) pdf.text("- Safe Actions", 20, 100);
            if (safeConditions) pdf.text("- Safe Conditions", 20, 110);

            pdf.text("Description:", 10, 120);
            pdf.text(description, 20, 130);

            pdf.text("Immediate Action:", 10, 160);
            pdf.text(immediateAction, 20, 170);

            pdf.text("Recommendations:", 10, 200);
            pdf.text(recommendations, 20, 210);

            pdf.save("Safety_Stop_Card.pdf");
        }
    </script>
